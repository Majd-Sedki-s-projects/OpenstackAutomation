#!/bin/bash
echo "telecoms144" | passwd --stdin root
printf "\033c" #clear screen

apt-get -y update
apt-get -y upgrade

echo "Installing vagrant"
apt-get -y install vagrant
echo "Installing ansible"
apt-get -y install ansible
echo "Installing git"
apt-get -y install git

echo "Create wordpress directory and cd into it."
mkdir ~/wordpress

cd ~/wordpress

echo "Cloning repo"
git clone https://github.com/Steven-N/Vagrant-Ansible-Wordpress.git

echo "Running vagrant"
vagrant up
